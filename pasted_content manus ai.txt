<script>

const items = document.querySelectorAll('.item');
const modal = document.getElementById('modal');
const modalContent = document.getElementById('modalContent');

let currentIndex = 0;
let filteredItems = [...items];

function showItem(index){
  const el = filteredItems[index];
  modalContent.innerHTML = el.dataset.type === 'video'
    ? `<video src="${el.dataset.src}" controls autoplay></video>`
    : `<img src="${el.dataset.src}">`;
}

items.forEach((el)=>{
  el.onclick=()=>{
    const category = el.dataset.id;
    filteredItems = [...items].filter(item => item.dataset.id === category);
    currentIndex = filteredItems.indexOf(el);

    modal.classList.add('active');
    showItem(currentIndex);
  }
});

document.querySelector('.next').onclick=()=>{
  currentIndex = (currentIndex+1)%filteredItems.length;
  showItem(currentIndex);
};

document.querySelector('.prev').onclick=()=>{
  currentIndex = (currentIndex-1+filteredItems.length)%filteredItems.length;
  showItem(currentIndex);
};

document.getElementById('closeModal').onclick=()=>{
  modal.classList.remove('active');
};

modal.onclick=(e)=>{
  if(e.target.id === 'modal'){
    modal.classList.remove('active');
  }
};

document.addEventListener('keydown',e=>{
  if(e.key === 'Escape') modal.classList.remove('active');
  if(e.key === 'ArrowRight') document.querySelector('.next').click();
  if(e.key === 'ArrowLeft') document.querySelector('.prev').click();
});

document.querySelectorAll('#filter-btns li').forEach(btn=>{
  btn.onclick=()=>{
    document.querySelectorAll('#filter-btns li').forEach(b=>b.classList.remove('active'));
    btn.classList.add('active');

    const target = btn.dataset.target;

    document.querySelectorAll('.item').forEach(item=>{
      item.style.display = (target==='all'||item.dataset.id===target)?'block':'none';
    });
  };
});

const toggle = document.getElementById('darkModeToggle');
toggle.onclick=()=>{
  document.body.classList.toggle('dark');
};

</script>
